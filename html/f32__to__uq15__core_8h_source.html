81JM_FORCEINLINE JM_HOT uint16_t
82f32_to_uq15_step(f32_to_uq15_state_t *state, float x)
83{
84 float s = state->scale * x;
85 /* Detect saturation before clamping; set sticky flag. */
86 state->clipped |= (uint8_t)(s > 32767.0f || s < -32768.0f);
87 s = fmaxf(s, -32768.0f);
88 s = fminf(s, 32767.0f);
89 int16_t v = (int16_t)lroundf(s);
90 /* Shift Q15 to offset-binary: 0 → 32768, -32768 → 0, 32767 → 65535 */
91 return (uint16_t)((int32_t)v + 32768);
92}
