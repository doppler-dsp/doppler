57static inline int64_t
58hsum_epi32_i64(__m256i v)
59{
60 __m128i lo128 = _mm256_extracti128_si256(v, 0);
61 __m128i hi128 = _mm256_extracti128_si256(v, 1);
62 /* Sign-extend all 8 int32 lanes to 8 × int64 (two 256-bit vectors). */
63 __m256i lo64 = _mm256_cvtepi32_epi64(lo128);
64 __m256i hi64 = _mm256_cvtepi32_epi64(hi128);
65 __m256i sum256 = _mm256_add_epi64(lo64, hi64); /* 4 × int64 */
66 __m128i slo = _mm256_extracti128_si256(sum256, 0);
67 __m128i shi = _mm256_extracti128_si256(sum256, 1);
68 __m128i r = _mm_add_epi64(slo, shi); /* 2 × int64 */
69 return (int64_t)_mm_extract_epi64(r, 0) + (int64_t)_mm_extract_epi64(r, 1);
70}

88static int64_t
89dot_q15_avx2(const int16_t *a, const int16_t *b, size_t n)
90{
91 int64_t result = 0;
92 size_t k = 0;
93
94 /* Flush to int64 after every 16-element block.
95 *
96 * A single _mm256_madd_epi16 call over all-Q15_MAX inputs produces
97 * int32 lanes of 2 × 32767² = 2,147,352,578 — only 131,069 below
98 * INT32_MAX. Accumulating two such calls overflows int32. We therefore
99 * reduce each block to int64 before moving on; the scalar tail handles
100 * any remainder. */
101 for (; k + 16 <= n; k += 16) {
102 __m256i va = _mm256_loadu_si256((const __m256i *)(a + k));
103 __m256i vb = _mm256_loadu_si256((const __m256i *)(b + k));
104 result += hsum_epi32_i64(_mm256_madd_epi16(va, vb));
105 }
106
107 for (; k < n; k++)
108 result += (int64_t)a[k] * b[k];
109
110 return result;
111}
