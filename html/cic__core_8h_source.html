111JM_FORCEINLINE JM_HOT size_t
112cic_decimate(cic_state_t *state, const float complex *in,
113 size_t n_in, float complex *out)
114{
115 const uint32_t R = state->R;
116 const uint32_t shift = state->shift;
117 size_t n_out = 0;
118
119 for (size_t i = 0; i < n_in; i++) {
120 /* CF32 → UQ16: saturate to Q15, shift to offset-binary [0, 65535]. */
121 float sr = crealf(in[i]) * 32768.0f;
122 float si = cimagf(in[i]) * 32768.0f;
123 if (sr > 32767.0f) sr = 32767.0f;
124 if (sr < -32768.0f) sr = -32768.0f;
125 if (si > 32767.0f) si = 32767.0f;
126 if (si < -32768.0f) si = -32768.0f;
127 uint64_t re = (uint64_t)((int32_t)(int16_t)sr + 32768);
128 uint64_t im = (uint64_t)((int32_t)(int16_t)si + 32768);
129
130 /* 4 integrators — unsigned wrap-around is intentional. */
131 re = state->integ_re[0] += re;
132 re = state->integ_re[1] += re;
133 re = state->integ_re[2] += re;
134 re = state->integ_re[3] += re;
135 im = state->integ_im[0] += im;
136 im = state->integ_im[1] += im;
137 im = state->integ_im[2] += im;
138 im = state->integ_im[3] += im;
139
140 if (++state->phase < R)
141 continue;
142 state->phase = 0;
143
144 /* 4 comb stages — M=1: y = x - prev; prev = x. */
145 uint64_t t;
146 t = state->comb_re[0]; state->comb_re[0] = re; re -= t;
147 t = state->comb_re[1]; state->comb_re[1] = re; re -= t;
148 t = state->comb_re[2]; state->comb_re[2] = re; re -= t;
149 t = state->comb_re[3]; state->comb_re[3] = re; re -= t;
150 t = state->comb_im[0]; state->comb_im[0] = im; im -= t;
151 t = state->comb_im[1]; state->comb_im[1] = im; im -= t;
152 t = state->comb_im[2]; state->comb_im[2] = im; im -= t;
153 t = state->comb_im[3]; state->comb_im[3] = im; im -= t;
154
155 /* UQ16 → CF32: right-shift to normalise, remove offset-binary bias. */
156 out[n_out++] = CMPLXF(
157 ((float)(uint16_t)(re >> shift) - 32768.0f) * (1.0f / 32768.0f),
158 ((float)(uint16_t)(im >> shift) - 32768.0f) * (1.0f / 32768.0f));
159 }
160 return n_out;
161}
