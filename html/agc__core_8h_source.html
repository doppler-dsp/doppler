161 agc_log10_ (double p)
162 {
163 uint64_t bits;
164 memcpy (&bits, &p, sizeof bits);
165 int e = (int)((bits >> 52) & 0x7FF) - 1023; /* p = m * 2^e */
166 bits = (bits & 0x000FFFFFFFFFFFFFULL) | 0x3FF0000000000000ULL;
167 double m;
168 memcpy (&m, &bits, sizeof m); /* m in [1, 2) */
169 /* log2(m) = (2/ln2) * (t + t^3/3 + ...), t = (m-1)/(m+1) in [0,1/3]. */
170 double t = (m - 1.0) / (m + 1.0);
171 double log2m = 2.885390081777927 * t * (1.0 + t * t * 0.3333333333333333);

239 JM_FORCEINLINE JM_HOT float complex
240 agc_step (agc_state_t *state, float complex x)
241 {
242 /* Stage 1: linear-in-dB gain. gain_db is voltage dB, so the linear
243 * multiplier is 10^(gain_db/20); 0.05 == 1/20. Record it so a
244 * following agc_steps() call ramps continuously from here. */
245 double g = agc_exp10_ (state->gain_db * 0.05);
246 state->g_last = g;
247 float complex y = x * (float)g;
248
249 /* Stage 2: power detector. Instantaneous output power folded into
250 * the EMA p_avg += alpha * (p - p_avg). */
251 double yr = (double)crealf (y);
252 double yi = (double)cimagf (y);
253 double p = yr * yr + yi * yi;
254 state->p_avg += state->alpha * (p - state->p_avg);
255
256 /* Stage 3: 1st-order loop filter. Integrate the dB error with step
257 * size 4*loop_bw (loop_bw is the loop noise bandwidth); the floor
258 * keeps log10 finite if p_avg has decayed to ~0 during silence. */
259 double meas_db = 10.0 * agc_log10_ (state->p_avg + AGC_POWER_FLOOR);
260 state->gain_db += 4.0 * state->loop_bw * (state->ref_db - meas_db);
261
262 /* Output clip — square clip (I and Q independent) to the
263 * programmable level, via the shared util primitive. Applied to
264 * the returned sample only; the detector above used the unclipped
265 * y, so the loop is unaffected. */
266 return square_clip (y, (float)agc_exp10_ (state->clip_db * 0.05));
267 }
