100JM_FORCEINLINE JM_HOT int64_t
101adc_step(adc_state_t *state, float x)
102{
103 double s = (double)x * state->scale;
104 if (state->dithering) {
105 /* TPDF dither: two xorshift32 draws summed → triangular PDF
106 * over [-1, +1] LSB. Each uniform is in [-0.5, +0.5] LSB. */
107 state->rng ^= state->rng << 13;
108 state->rng ^= state->rng >> 17;
109 state->rng ^= state->rng << 5;
110 uint32_t r2 = state->rng * 2654435761u;
111 s += ((double)(int32_t)state->rng + (double)(int32_t)r2)
112 * (1.0 / 4294967296.0);
113 }
114 state->clipped |= (uint8_t)(s > (double)state->clip_max
115 || s < (double)state->clip_min);
116 int64_t v = llround(s);
117 if (v > state->clip_max) v = state->clip_max;
118 if (v < state->clip_min) v = state->clip_min;
119 return (int64_t)v;
120}
