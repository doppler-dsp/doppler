114JM_FORCEINLINE JM_HOT float complex
115synth_step(synth_state_t *state)
116{
117 if (state->wtype >= SYNTH_PN) {
118 if (state->sym_pos == 0) {
119 if (state->wtype == SYNTH_QPSK) {
120 uint8_t b0 = pn_step(state->pn);
121 uint8_t b1 = pn_step(state->pn);
122 const float s = 0.70710678118654752f;
123 state->cur_re = b0 ? -s : s;
124 state->cur_im = b1 ? -s : s;
125 } else { /* pn or bpsk: +-1 */
126 uint8_t b = pn_step(state->pn);
127 state->cur_re = b ? -1.0f : 1.0f;
128 state->cur_im = 0.0f;
129 }
130 }
131 if (++state->sym_pos >= state->nsps)
132 state->sym_pos = 0;
133 }
134 float complex sym = state->cur_re + state->cur_im * I;
135 float complex carrier = 1.0f + 0.0f * I;
136 if (state->lo)
137 lo_steps(state->lo, 1, &carrier);
138 float complex noise = 0.0f + 0.0f * I;
139 if (state->awgn)
140 awgn_generate(state->awgn, 1, &noise);
141 return sym * carrier + noise;
142}
